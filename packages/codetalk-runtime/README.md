## Install

```bash
npm install --save codetalk-runtime
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'codetalk-runtime'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
